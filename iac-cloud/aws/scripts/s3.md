# aws de s3 bucketlerin siyahisi ve storage ucun script

```bash
for bucket in $(aws s3api list-buckets --query "Buckets[*].Name" --output text); do
    echo "Bucket: $bucket"
    aws s3 ls s3://$bucket --recursive --human-readable --summarize | grep "Total Objects\|Total Size"
    echo "--------------------------------------"
done
```
